# 대시보드 컴포넌트 구현 가이드

## 개요
특허 AI 플랫폼의 대시보드를 미니멀하고 고급스러운 디자인으로 재구현하기 위한 상세한 컴포넌트 가이드입니다.
React, TypeScript, Tailwind CSS를 사용하여 현대적이고 반응형 대시보드를 구축합니다.

### 2. 통계 카드 컴포넌트
```tsx
// src/pages/Dashboard/components/StatsCard.tsx

import React from 'react';
import { motion } from 'framer-motion';
import { ArrowUpIcon, ArrowDownIcon, MinusIcon } from '@heroicons/react/24/solid';
import { StatCardData } from '../types/dashboard.types';

interface StatsCardProps extends StatCardData {
  className?: string;
  onClick?: () => void;
}

const StatsCard: React.FC<StatsCardProps> = ({
  title,
  value,
  change,
  icon,
  trend,
  status = 'good',
  className = '',
  onClick
}) => {
  const getChangeIcon = () => {
    if (!change) return null;
    
    switch (change.type) {
      case 'increase':
        return <ArrowUpIcon className="w-3 h-3" />;
      case 'decrease':
        return <ArrowDownIcon className="w-3 h-3" />;
      default:
        return <MinusIcon className="w-3 h-3" />;
    }
  };

  const getChangeColor = () => {
    if (!change) return '';
    
    switch (change.type) {
      case 'increase':
        return 'bg-green-100 text-green-800';
      case 'decrease':
        return 'bg-red-100 text-red-800';
      default:
        return 'bg-neutral-100 text-neutral-800';
    }
  };

  const getStatusColor = () => {
    switch (status) {
      case 'warning':
        return 'border-l-amber-500';
      case 'critical':
        return 'border-l-red-500';
      default:
        return 'border-l-blue-500';
    }
  };

  return (
    <motion.div
      whileHover={{ y: -2, boxShadow: '0 10px 25px -5px rgba(0, 0, 0, 0.1)' }}
      whileTap={{ scale: 0.98 }}
      className={`
        bg-white rounded-xl shadow-sm border border-neutral-200 border-l-4 ${getStatusColor()}
        p-6 cursor-pointer transition-all duration-200 hover:shadow-md
        ${className}
      `}
      onClick={onClick}
    >
      <div className="flex items-center justify-between">
        <div className="flex items-center space-x-3">
          {icon && (
            <div className="p-2 bg-neutral-50 rounded-lg">
              {icon}
            </div>
          )}
          <div>
            <p className="text-sm font-medium text-neutral-600 mb-1">{title}</p>
            <p className="text-2xl font-semibold text-neutral-900">
              {typeof value === 'number' ? value.toLocaleString() : value}
            </p>
          </div>
        </div>

        {change && (
          <div className={`
            flex items-center space-x-1 px-2 py-1 rounded-full text-xs font-medium
            ${getChangeColor()}
          `}>
            {getChangeIcon()}
            <span>{Math.abs(change.value)}%</span>
          </div>
        )}
      </div>

      {change && (
        <p className="text-xs text-neutral-500 mt-3">{change.period}</p>
      )}

      {/* Mini trend chart */}
      {trend && trend.length > 0 && (
        <div className="mt-4 h-8">
          <svg className="w-full h-full" viewBox="0 0 100 20">
            <polyline
              fill="none"
              stroke="currentColor"
              strokeWidth="1.5"
              className="text-blue-500"
              points={trend.map((point, index) => 
                `${(index / (trend.length - 1)) * 100},${20 - (point.value / Math.max(...trend.map(p => p.value))) * 15}`
              ).join(' ')}
            />
          </svg>
        </div>
      )}
    </motion.div>
  );
};

export default StatsCard;
```

```tsx
// src/pages/Dashboard/components/ActivityFeed.tsx

import React from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { 
  MagnifyingGlassIcon,
  DocumentTextIcon,
  UserIcon,
  ArrowDownTrayIcon,
  ShareIcon
} from '@heroicons/react/24/outline';
import { RecentActivity } from '../types/dashboard.types';

interface ActivityFeedProps {
  activities: RecentActivity[];
  maxItems?: number;
  className?: string;
}

const ActivityFeed: React.FC<ActivityFeedProps> = ({
  activities,
  maxItems = 10,
  className = ''
}) => {
  const getActivityIcon = (type: RecentActivity['type']) => {
    const iconClass = "w-4 h-4";
    
    switch (type) {
      case 'search':
        return <MagnifyingGlassIcon className={iconClass} />;
      case 'report':
        return <DocumentTextIcon className={iconClass} />;
      case 'login':
        return <UserIcon className={iconClass} />;
      case 'download':
        return <ArrowDownTrayIcon className={iconClass} />;
      case 'share':
        return <ShareIcon className={iconClass} />;
      default:
        return <div className={`${iconClass} bg-neutral-300 rounded-full`} />;
    }
  };

  const getStatusColor = (status: RecentActivity['status']) => {
    switch (status) {
      case 'success':
      case 'completed':
        return 'bg-green-500';
      case 'pending':
        return 'bg-amber-500';
      case 'error':
        return 'bg-red-500';
      default:
        return 'bg-neutral-400';
    }
  };

  const getActivityTypeLabel = (type: RecentActivity['type']) => {
    switch (type) {
      case 'search':
        return '검색';
      case 'report':
        return '보고서';
      case 'login':
        return '로그인';
      case 'download':
        return '다운로드';
      case 'share':
        return '공유';
      default:
        return '활동';
    }
  };

  const formatTimestamp = (timestamp: string) => {
    const date = new Date(timestamp);
    const now = new Date();
    const diffInMinutes = Math.floor((now.getTime() - date.getTime()) / (1000 * 60));
    
    if (diffInMinutes < 1) return '방금 전';
    if (diffInMinutes < 60) return `${diffInMinutes}분 전`;
    if (diffInMinutes < 1440) return `${Math.floor(diffInMinutes / 60)}시간 전`;
    return `${Math.floor(diffInMinutes / 1440)}일 전`;
  };

  const displayActivities = activities.slice(0, maxItems);

  return (
    <div className={`bg-white rounded-xl shadow-sm border border-neutral-200 p-6 ${className}`}>
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-lg font-semibold text-neutral-900">최근 활동</h3>
        <span className="text-sm text-neutral-500">{activities.length}개 활동</span>
      </div>

      <div className="space-y-3 max-h-96 overflow-y-auto">
        <AnimatePresence>
          {displayActivities.map((activity, index) => (
            <motion.div
              key={activity.id}
              initial={{ opacity: 0, x: -20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: 20 }}
              transition={{ delay: index * 0.05 }}
              className="flex items-start space-x-3 p-3 rounded-lg hover:bg-neutral-50 transition-colors cursor-pointer group"
            >
              {/* Status Indicator */}
              <div className="flex items-center space-x-2 mt-1">
                <div className={`w-2 h-2 rounded-full ${getStatusColor(activity.status)}`} />
                <div className="p-1.5 bg-neutral-100 rounded-lg group-hover:bg-neutral-200 transition-colors">
                  {getActivityIcon(activity.type)}
                </div>
              </div>

              {/* Content */}

                <div className="flex items-center space-x-2 mb-1">
                  <span className="text-xs font-medium text-neutral-500 bg-neutral-100 px-2 py-0.5 rounded-full">
                    {getActivityTypeLabel(activity.type)}
                  </span>

                    {formatTimestamp(activity.timestamp)}
                  </span>

                </p>

                  </p>
                )}
              </div>

                  ? 'bg-green-100 text-green-800' 
                  : activity.status === 'pending'
                  ? 'bg-amber-100 text-amber-800'
                  : activity.status === 'error'
                  ? 'bg-red-100 text-red-800'
                  : 'bg-neutral-100 text-neutral-800'
                }
              `}>
                {activity.status === 'success' || activity.status === 'completed' ? '완료' :
                 activity.status === 'pending' ? '진행중' :
                 activity.status === 'error' ? '오류' : '알 수 없음'}
              </div>
            </motion.div>
          ))}
        </AnimatePresence>

        {displayActivities.length === 0 && (
          <div className="text-center py-8">
            <p className="text-neutral-500 text-sm">최근 활동이 없습니다</p>
          </div>
        )}
      </div>

      {activities.length > maxItems && (
        <div className="mt-4 pt-4 border-t border-neutral-200">
          <button className="w-full text-sm text-blue-600 hover:text-blue-700 font-medium">
            모든 활동 보기 ({activities.length - maxItems}개 더)
          </button>
        </div>
      )}
    </div>
  );
};
```

```tsx
// src/pages/Dashboard/components/DonutChart.tsx

import React, { useMemo } from 'react';
import { PieChart, Pie, Cell, ResponsiveContainer, Tooltip } from 'recharts';
import { ChartDataPoint } from '../types/dashboard.types';

interface DonutChartProps {
  data: ChartDataPoint[];
  centerText?: string;
  showLegend?: boolean;
  innerRadius?: number;
  outerRadius?: number;
  height?: number;
  className?: string;
}

const DonutChart: React.FC<DonutChartProps> = ({
  data,
  centerText,
  showLegend = false,
  innerRadius = 60,
  outerRadius = 100,
  height = 320,
  className = ''
}) => {
  const colors = [
    '#3b82f6', '#10b981', '#f59e0b', '#ef4444', '#8b5cf6',
    '#06b6d4', '#84cc16', '#f97316', '#ec4899', '#6366f1'
  ];

  const chartData = useMemo(() => {
    return data.map((item, index) => ({
      name: item.x,
      value: item.y,
      color: item.color || colors[index % colors.length]
    }));
  }, [data]);

  const total = useMemo(() => {
    return chartData.reduce((sum, item) => sum + item.value, 0);
  }, [chartData]);

  const CustomTooltip = ({ active, payload }: any) => {
    if (active && payload && payload.length) {
      const data = payload[0];
      const percentage = ((data.value / total) * 100).toFixed(1);
      
      return (
        <div className="bg-white p-3 border border-neutral-200 rounded-lg shadow-lg">
          <p className="text-sm font-medium text-neutral-900">{data.name}</p>
          <p className="text-sm text-neutral-600">
            값: <span className="font-semibold">{data.value}</span>
          </p>
          <p className="text-xs text-neutral-500">비율: {percentage}%</p>
        </div>
      );
    }
    return null;
  };

  const renderCustomLabel = ({ cx, cy, midAngle, innerRadius, outerRadius, percent }: any) => {
    if (percent < 0.05) return null; // 5% 미만은 라벨 숨김
    
    const RADIAN = Math.PI / 180;
    const radius = innerRadius + (outerRadius - innerRadius) * 0.5;
    const x = cx + radius * Math.cos(-midAngle * RADIAN);
    const y = cy + radius * Math.sin(-midAngle * RADIAN);

    return (
      <text 
        x={x} 
        y={y} 
        fill="white" 
        textAnchor={x > cx ? 'start' : 'end'} 
        dominantBaseline="central"
        fontSize={12}
        fontWeight="medium"
      >
        {`${(percent * 100).toFixed(0)}%`}
      </text>
    );
  };

  return (
    <div className={`w-full ${className}`} style={{ height }}>
      <div className="relative">
        <ResponsiveContainer width="100%" height="100%">
          <PieChart>
            <Pie
              data={chartData}
              cx="50%"
              cy="50%"
              labelLine={false}
              label={renderCustomLabel}
              outerRadius={outerRadius}
              innerRadius={innerRadius}
              fill="#8884d8"
              dataKey="value"
              animationBegin={0}
              animationDuration={800}
            >
              {chartData.map((entry, index) => (
                <Cell key={`cell-${index}`} fill={entry.color} />
              ))}
            </Pie>
            <Tooltip content={<CustomTooltip />} />
          </PieChart>
        </ResponsiveContainer>

        {/* Center Text */}
        {centerText && (
          <div className="absolute inset-0 flex items-center justify-center pointer-events-none">
            <div className="text-center">
              <p className="text-sm font-medium text-neutral-600">{centerText}</p>
              <p className="text-lg font-semibold text-neutral-900">{total.toLocaleString()}</p>
            </div>
          </div>
        )}
      </div>

      {/* Legend */}
      {showLegend && (
        <div className="mt-4 grid grid-cols-2 gap-2">
          {chartData.map((item, index) => (
            <div key={index} className="flex items-center space-x-2">
              <div 
                className="w-3 h-3 rounded-full" 
                style={{ backgroundColor: item.color }}
              />
              <span className="text-xs text-neutral-600 truncate">{item.name}</span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
};

export default DonutChart;
```

```tsx
// src/pages/Dashboard/hooks/useDashboardData.ts

import { useState, useEffect, useCallback } from 'react';
import { DashboardStats } from '../types/dashboard.types';
import { getDashboardStats } from '../../../lib/api/dashboard';
import { useAuthStore } from '../../../store/authStore';

interface UseDashboardDataReturn {
  data: DashboardStats | null;
  loading: boolean;
  error: string | null;
  refetch: () => Promise<void>;
}

export const useDashboardData = (period: '7d' | '30d' | '90d' = '30d'): UseDashboardDataReturn => {
  const [data, setData] = useState<DashboardStats | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const { user } = useAuthStore();

  const fetchData = useCallback(async () => {
    if (!user) {
      setError('사용자 인증이 필요합니다');
      setLoading(false);
      return;
    }

    try {
      setLoading(true);
      setError(null);
      
      const result = await getDashboardStats(period);
      setData(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : '데이터를 불러오는 중 오류가 발생했습니다');
    } finally {
      setLoading(false);
    }
  }, [user, period]);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  const refetch = useCallback(async () => {
    await fetchData();
  }, [fetchData]);

  return { data, loading, error, refetch };
};
```

```tsx
// src/pages/Dashboard/hooks/useRealTimeStats.ts

import { useState, useEffect, useRef } from 'react';
import { useAuthStore } from '../../../store/authStore';

interface RealTimeStats {
  active_sessions: number;
  total_logins: number;
  last_updated: string;
}

interface UseRealTimeStatsReturn {
  realTimeStats: Partial<RealTimeStats>;
  isConnected: boolean;
  error: string | null;
}

export const useRealTimeStats = (): UseRealTimeStatsReturn => {
  const [realTimeStats, setRealTimeStats] = useState<Partial<RealTimeStats>>({});
  const [isConnected, setIsConnected] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const { user } = useAuthStore();
  const intervalRef = useRef<NodeJS.Timeout | null>(null);

  useEffect(() => {
    if (!user) return;

    const fetchRealTimeStats = async () => {
      try {
        // 실제 구현에서는 WebSocket 또는 Server-Sent Events 사용
        // 여기서는 폴링으로 시뮬레이션
        const response = await fetch('/api/dashboard/realtime-stats', {
          headers: {
            'Authorization': `Bearer ${user.access_token}`
          }
        });

        if (response.ok) {
          const data = await response.json();
          setRealTimeStats(data);
          setIsConnected(true);
          setError(null);
        } else {
          throw new Error('실시간 데이터를 가져올 수 없습니다');
        }
      } catch (err) {
        setError(err instanceof Error ? err.message : '연결 오류');
        setIsConnected(false);
      }
    };

    // 초기 데이터 로드
    fetchRealTimeStats();

    // 30초마다 업데이트
    intervalRef.current = setInterval(fetchRealTimeStats, 30000);

    return () => {
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
      }
    };
  }, [user]);

  return { realTimeStats, isConnected, error };
};
```
